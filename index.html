ลำดับ ตำบล หมู่ที่ / ชื่อหมู่บ้าน จำนวน อช. (ช/ญ) วันที่บันทึกล่าสุด การจัดการ

ตำบล ........................ อำเภอห้วยราช จังหวัดบุรีรัมย์
หมู่ที่ ........ บ้าน ........................
หมายเหตุ:
  1. ส่งเอกสารใบสมัครอาสาพัฒนาชุมชน (อช.) เฉพาะผู้ที่ได้รับคัดเลือกฯ
  2. ส่งแบบตรวจสอบคุณสมบัติของผู้ที่ได้รับคัดเลือกฯ
กรอก Web App URL เพื่อเปิดใช้งานการบันทึกไปยัง Google Sheets

เมื่อคุณกด "การทำให้ใช้งานได้" (Deploy) ใน Google Apps Script แล้ว ให้นำ URL ที่ได้มาวางในช่องด้านล่างนี้ ระบบจะทำการส่งข้อมูลไปยัง Google Sheets โดยอัตโนมัติเมื่อมีการบันทึกรายงาน

วิธีเชื่อมต่อแบบฟอร์มนี้เข้ากับ Google Sheets (โดยผู้กรอกไม่ต้อง Sign-in)

1. เปิด Google Sheets ของคุณ แล้วไปที่เมนู ส่วนขยาย (Extensions) > Apps Script

2. ลบโค้ดเดิมออกทั้งหมด แล้ววางโค้ดสคริปต์ด้านล่างนี้ลงไป:

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "Timestamp", "ตำบล", "หมู่ที่", "ชื่อหมู่บ้าน", "จำนวน อช.", 
        "อช.1 คำนำหน้า", "อช.1 ชื่อ-สกุล", "อช.1 เพศ", "อช.1 อายุ", "อช.1 เบอร์โทร",
        "อช.2 คำนำหน้า", "อช.2 ชื่อ-สกุล", "อช.2 เพศ", "อช.2 อายุ", "อช.2 เบอร์โทร",
        "อช.3 คำนำหน้า", "อช.3 ชื่อ-สกุล", "อช.3 เพศ", "อช.3 อายุ", "อช.3 เบอร์โทร",
        "อช.4 คำนำหน้า", "อช.4 ชื่อ-สกุล", "อช.4 เพศ", "อช.4 อายุ", "อช.4 เบอร์โทร",
        "อช.5 คำนำหน้า", "อช.5 ชื่อ-สกุล", "อช.5 เพศ", "อช.5 อายุ", "อช.5 เบอร์โทร",
        "อช.6 คำนำหน้า", "อช.6 ชื่อ-สกุล", "อช.6 เพศ", "อช.6 อายุ", "อช.6 เบอร์โทร"
      ]);
    }

    var row = [
      new Date(), data.tambon, data.moo, data.villageName, data.volunteers.length
    ];

    for (var i = 0; i < 6; i++) {
      if (i < data.volunteers.length) {
        var v = data.volunteers[i];
        row.push(v.prefix, v.fullname, v.gender, v.age, v.phone);
      } else {
        row.push("", "", "", "", "");
      }
    }

    sheet.appendRow(row);
    return ContentService.createTextOutput(JSON.stringify({"result": "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({"result": "error", "error": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

3. กดปุ่ม "การทำให้ใช้งานได้" (Deploy) > "การทำให้ใช้งานได้รายการใหม่" (New Deployment)

4. เลือกประเภทเป็น "เว็บแอป" (Web app), กำหนด "ผู้ที่มีสิทธิ์เข้าถึง" (Who has access) เป็น "ทุกคน" (Anyone) แล้วกด Deploy จากนั้นคัดลอก Web App URL นำมาวางในช่องด้านบนได้ทันที
