วิธีแก้ปัญหา & การเชื่อมต่อ Google Sheets

ขั้นตอนการนำโค้ด Apps Script ไปติดตั้งใน Google Sheets เพื่อรับข้อมูลเข้าตารางโดยอัตโนมัติ

ขั้นตอนการติดตั้ง Google Apps Script ใน Google Sheets (ทำครั้งเดียว):

  1. เปิดไฟล์ Google Sheets ของคุณ: คลิกที่นี่เพื่อเปิดสเปรดชีต
  2. ไปที่เมนูด้านบนเลือก ส่วนขยาย (Extensions) > Apps Script
  3. ลบโค้ดเดิมทั้งหมดในไฟล์ Code.gs ออก แล้ววาง **โค้ดสคริปต์รับข้อมูล** ด้านล่างนี้ลงไป
  4. กดปุ่ม **บันทึก (Save)** รูปแผ่นดิสก์
  5. กดปุ่ม **ทำให้ใช้งานได้อย่างรวดเร็ว (Deploy)** มุมขวาบน > เลือก **การทำให้ใช้งานได้ใหม่ (New deployment)**
  6. ในช่อง ผู้ที่มีสิทธิ์เข้าถึง (Who has access) ให้เปลี่ยนเป็น **"ทุกคน (Anyone)"** (สำคัญมาก!)
  7. กด **ทำให้ใช้งานได้ (Deploy)** แล้วคัดลอก **URL ของเว็บแอป (Web App URL)** นำมาวางในช่องด้านล่าง
โค้ดสำหรับวางใน Apps Script (Code.gs) 
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var data = JSON.parse(e.postData.contents);
    
    // หากยังไม่มีหัวตาราง ให้สร้างหัวตารางอัตโนมัติ
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "วันที่บันทึก", "ตำบล", "หมู่บ้าน", 
        "อช.1 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร",
        "อช.2 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร",
        "อช.3 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร",
        "อช.4 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร",
        "อช.5 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร",
        "อช.6 ชื่อ-สกุล", "เลขบัตรประชาชน", "บ้านเลขที่", "อาชีพ", "เบอร์โทร"
      ]);
      sheet.getRange(1, 1, 1, 33).setFontWeight("bold").setBackground("#d9e1f2");
    }

    var row = [data.updatedAt, data.tambon, data.village];
    
    // ใส่ข้อมูล อช. แต่ละคน (พร้อมฝังเครื่องหมาย ' เพื่อไม่ให้เลข 0 และเลข 13 หลักสูญหาย)
    for (var i = 0; i < 6; i++) {
      if (data.volunteers && data.volunteers[i]) {
        var v = data.volunteers[i];
        row.push(
          v.name || "", 
          "'" + (v.idcard || ""), 
          v.house_no || "", 
          v.occupation || "", 
          "'" + (v.phone || "")
        );
      } else {
        row.push("", "", "", "", "");
      }
    }
    
    sheet.appendRow(row);
    
    return ContentService.createTextOutput(JSON.stringify({"status": "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
