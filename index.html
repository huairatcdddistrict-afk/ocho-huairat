คุณสามารถวาง Direct URL หรือ Raw Link รูปภาพโลโก้ของคุณจาก GitHub เพื่อนำมาแสดงในระบบและใบพิมพ์ A4

กรอก Web App URL เพื่อเปิดใช้งานการบันทึกไปยัง Google Sheets

เมื่อคุณกด "การทำให้ใช้งานได้" (Deploy) ใน Google Apps Script แล้ว ให้นำ URL ที่ได้มาวางในช่องด้านล่างนี้ ระบบจะทำการส่งข้อมูลไปยัง Google Sheets โดยอัตโนมัติเมื่อมีการกดบันทึก

วิธีเชื่อมต่อแบบฟอร์มนี้เข้ากับ Google Sheets (โดยผู้กรอกไม่ต้อง Sign-in)

ขั้นตอนที่ 1: สร้าง Google Sheets & เปิด Apps Script

1. สร้าง Google Sheets ใหม่ใน Google Drive ของคุณ

2. ไปที่เมนู ส่วนขยาย (Extensions) → Apps Script

ขั้นตอนที่ 2: คัดลอกโค้ด Google Apps Script (Code.gs) 
function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    
    // ตั้งค่าหัวตารางถ้ายังไม่มี
    if (sheet.getLastRow() === 0) {
      sheet.appendRow([
        "วัน-เวลาบันทึก", "ตำบล", "หมู่ที่", "ชื่อหมู่บ้าน",
        "คนที่ 1 ชื่อ-สกุล", "เลขบัตรประชาชน 1", "บ้านเลขที่ 1", "อาชีพ 1", "โทรศัพท์ 1",
        "คนที่ 2 ชื่อ-สกุล", "เลขบัตรประชาชน 2", "บ้านเลขที่ 2", "อาชีพ 2", "โทรศัพท์ 2",
        "คนที่ 3 ชื่อ-สกุล", "เลขบัตรประชาชน 3", "บ้านเลขที่ 3", "อาชีพ 3", "โทรศัพท์ 3",
        "คนที่ 4 ชื่อ-สกุล", "เลขบัตรประชาชน 4", "บ้านเลขที่ 4", "อาชีพ 4", "โทรศัพท์ 4",
        "คนที่ 5 ชื่อ-สกุล", "เลขบัตรประชาชน 5", "บ้านเลขที่ 5", "อาชีพ 5", "โทรศัพท์ 5",
        "คนที่ 6 ชื่อ-สกุล", "เลขบัตรประชาชน 6", "บ้านเลขที่ 6", "อาชีพ 6", "โทรศัพท์ 6"
      ]);
    }
    
    var data = JSON.parse(e.postData.contents);
    
    var row = [
      new Date(), data.tambon, data.moo, data.villageName
    ];
    
    for (var i = 0; i < 6; i++) {
      if (data.volunteers && data.volunteers[i]) {
        var v = data.volunteers[i];
        row.push(v.prefix + v.fullname, v.idCard, v.houseNo, v.occupation, v.phone);
      } else {
        row.push("", "", "", "", "");
      }
    }
    
    sheet.appendRow(row);
    
    return ContentService.createTextOutput(JSON.stringify({ "result": "success" }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ "result": "error", "error": err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
ขั้นตอนที่ 3: Deploy เป็น Web App ให้ทุกคนกรอกได้แบบไม่ต้อง Sign-in

1. กดปุ่ม การทำให้ใช้งานได้ (Deploy) → การทำให้ใช้งานได้ใหม่ (New Deployment)

2. เลือกประเภทเป็น เว็บแอป (Web App)

3. ตั้งค่า ผู้มีสิทธิ์เข้าถึง (Who has access): เลือกเป็น ทุกคน (Anyone)

4. กดปุ่มทำให้ใช้งานได้ คัดลอก Web App URL ที่ได้นำมาวางในช่องด้านบน
